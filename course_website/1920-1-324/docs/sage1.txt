print("Section 1.1 of Elementary Linear Algebra by Larson")

#To solve a system of equations, tell Sage each of the equations, using ==, then what variables to solve for
###
# Example 2
###
# Begin by Letting Sage know that we are using these letters as mathematical variable names.
x1,x2 = var('x1,x2')
sol2 = solve([x1+2*x2 == 4],[x1,x2])
print("The answer to Example 2 is")
print(sol2)
# Notice that Sage is setting x2 to be the free variable with parameter r1.
# What do you think would happen if you used solve([x1+2*x2 == 4],[x2,x1]) ?

###
# Example 3
###
x,y,z = var('x,y,z')
sol3 = solve([3*x+2*y-z == 3],[x,y,z])
print("The answer to Example 3 is")
print(sol3)


###
# Example 4
###
x,y = var('x,y')
sol4a = solve([x+y==3, x-y==-1],[x,y])
print("The answer to Example 4a is")
print(sol4a)

sol4b = solve([x+y==3, 2*x+2*y==6],[x,y])
print("The answer to Example 4b is")
print(sol4b)


sol4c = solve([x+y==3, x+y==1],[x,y])
print("The answer to Example 4c is")
print(sol4c)

# To plot the lines, we use the plot function after solving for y.
plot([3-x, x+1],[x,-1.5,3.5]).show(title = "Example 4a")
plot([3-x, 1/2*(6-2*x)],[x,-1.5,3.5]).show(title = "Example 4b")
plot([3-x, 1-x],[x,-1.5,3.5]).show(title = "Example 4c")
